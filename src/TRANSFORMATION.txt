    Point rotationAroundX(Point point,double angle){
        double x=point.getX();
        double y=point.getY();
        double z=point.getZ();
        return new Point(
                x,
                y*Math.cos(angle)-z*Math.sin(angle),
                y*Math.sin(angle)+z*Math.cos(angle)
        );
    }
    Point rotationAroundY(Point point,double angle){
        double x=point.getX();
        double y=point.getY();
        double z=point.getZ();
        return new Point(
                x*Math.cos(angle)+z*Math.sin(angle),
                y,
                -x*Math.sin(angle)+z*Math.cos(angle)
        );
    }

    Point rotationAroundZ(Point point,double angle){
        double x=point.getX();
        double y=point.getY();
        double z=point.getZ();
        return new Point(
                x*Math.cos(angle)-y*Math.sin(angle),
                x*Math.sin(angle)+y*Math.cos(angle),
                z
        );
    }
